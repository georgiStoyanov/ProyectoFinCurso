------------------------------------------------------------------Notas---------------------------------------------------------------
Los posibles servicios se pueden añadir al mismo fieldset con checkboxes y el segundo paso sera especificar los metros cuadrados o lineales de los servicios seleccionados.
El tercer paso sera calcular el presupuesto y el cuarto paso sera la informacion de contanto y la peticion de inspeccion visual de la obra.
Se puede hacer un boton que si falte algun servicio deseado que se calcule el presupuesto y que se mande una descripcion de lo que se va pedir hacer.

Lo que haran los botones del menu sera solo redirigir a la página de presupuestos con la categoria ya seleccionada.

//comentarios_inicio
	create table comentarios_inicio(
		id int auto_increment,
		email varchar(500),
		name varchar(200),
		content varchar(10000),
		primary key (id)
	);
	
//usuarios
	create table usuarios(
		id int auto_increment,
		email varchar(100) UNIQUE,
		pass varchar(500),
		derechos varchar(50),
		primary key(id)
	)
